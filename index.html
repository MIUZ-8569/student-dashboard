                console.error('Load error:', e);

                setSync('err');

            }

        }


        async function saveUserData() {

            if (!userDocRef || !currentUser) return;

            setSync('warn');

            try {

                await userDocRef.set({

                    tasks,

                    finances,

                    goals,

                    resources,

                    health: healthData,

                    focus: focusData,

                    water,

                    updatedAt: firebase.firestore.FieldValue.serverTimestamp(),

                    email: currentUser.email,

                    name: currentUser.displayName

                }, { merge: true });

                setSync('ok');

            } catch (e) {

                console.error('Save error:', e);

                setSync('err');

            }

        }


        function setSync(state) {

        function autoSave() {

            if (saveTimeout) clearTimeout(saveTimeout);

            saveTimeout = setTimeout(() => {

                saveUserData();

                saveTimeout = null;

            }, 800);

        }


        function setupAutoSave() {

            // Intercept all data-changing functions

            const origRender = renderAll;

            renderAll = function() {

                origRender();

                autoSave();

            };

            // Also save on input changes

            document.querySelectorAll('input, select, textarea').forEach(el => {

                el.addEventListener('change', autoSave);

                el.addEventListener('input', autoSave);

            });

        }

        function pad(n) { return n < 10 ? '0' + n : '' + n; }


        function format12h(date) {

            let h = date.getHours(),

                m = date.getMinutes(),

                s = date.getSeconds();

            const ampm = h >= 12 ? 'PM' : 'AM';

            let h12 = h % 12;

            if (h12 === 0) h12 = 12;

            return pad(h12) + ':' + pad(m) + ':' + pad(s) + ' ' + ampm;

        }

        function dayOfYear(d) {

            const start = new Date(d.getFullYear(), 0, 0);

            const diff = (d - start) + ((start.getTimezoneOffset() - d.getTimezoneOffset()) * 60000);

            return Math.floor(diff / 86400000);

        }


        function isLeap(y) { return (y % 4 === 0 && y % 100 !== 0) || y % 400 === 0; }

            }

        function updatePrayerTimes() {

            const now = new Date();

            const dateStr = now.getFullYear() + '-' + String(now.getMonth() + 1).padStart(2, '0') + '-' + String(now.getDate())

                .padStart(2, '0');


            // Simple calculation based on Dhaka (approximate)

            // Using a fixed offset method for demonstration

            const lat = 23.8103,

                lng = 90.4125;

            const prayerTimes = calculatePrayerTimes(now, lat, lng);


            const grid = document.getElementById('prayer-grid');

            grid.innerHTML = '';

            let nextPrayer = '';

            let nextTime = Infinity;


            const names = ['Fajr', 'Dhuhr', 'Asr', 'Maghrib', 'Isha'];

            const times = [prayerTimes.fajr, prayerTimes.dhuhr, prayerTimes.asr, prayerTimes.maghrib, prayerTimes.isha];


            names.forEach((name, i) => {

                const timeStr = times[i];

                const [h, m] = timeStr.split(':').map(Number);

                const prayerDate = new Date(now);

                prayerDate.setHours(h, m, 0);

                const diff = prayerDate - now;

                const isCurrent = diff > 0 && diff < 3600000;


                const div = document.createElement('div');

                div.className = 'prayer-item' + (isCurrent ? ' current' : '');

                div.innerHTML = `<div class="p-name">${name}</div><div class="p-time">${timeStr}</div>`;

                grid.appendChild(div);


                if (diff > 0 && diff < nextTime) {

                    nextTime = diff;

                    nextPrayer = `${name} at ${timeStr}`;

                }

            });


            // If all prayers passed, show next day's Fajr

            if (!nextPrayer) {

                const tomorrow = new Date(now);

                tomorrow.setDate(tomorrow.getDate() + 1);

                const nextFajr = calculatePrayerTimes(tomorrow, lat, lng).fajr;

                nextPrayer = `Fajr (tomorrow) at ${nextFajr}`;

            }


            document.getElementById('next-prayer').textContent = nextPrayer;

            document.getElementById('prayer-location').textContent = 'Dhaka, BD';

        }


        // Simplified prayer time calculator (approximate)

        function calculatePrayerTimes(date, lat, lng) {

            // This is a simplified approximation - for production use a proper API

            const day = date.getDate();

            const month = date.getMonth() + 1;

            const year = date.getFullYear();


            // Approximate times for Dhaka (these are rough estimates)

            // In production, use: https://api.aladhan.com/v1/timingsByCity

            const baseTimes = {

                fajr: '05:00',

                dhuhr: '12:00',

                asr: '15:30',

                maghrib: '18:00',

                isha: '19:30'

            };


            // Adjust based on day of year

            const dayOfYear = Math.floor((date - new Date(date.getFullYear(), 0, 0)) / 86400000);

            const seasonalAdjust = Math.sin((dayOfYear - 80) * Math.PI / 180) * 30;


            const adjustTime = (timeStr, minutes) => {

                const [h, m] = timeStr.split(':').map(Number);

                const total = h * 60 + m + Math.round(minutes);

                const nh = Math.floor(total / 60) % 24;

                const nm = total % 60;

                return String(nh).padStart(2, '0') + ':' + String(nm).padStart(2, '0');

            };


            return {

                fajr: adjustTime(baseTimes.fajr, -seasonalAdjust * 0.2),

                dhuhr: adjustTime(baseTimes.dhuhr, 0),

                asr: adjustTime(baseTimes.asr, seasonalAdjust * 0.1),

                maghrib: adjustTime(baseTimes.maghrib, seasonalAdjust * 0.15),

                isha: adjustTime(baseTimes.isha, seasonalAdjust * 0.1)

            };

        }
